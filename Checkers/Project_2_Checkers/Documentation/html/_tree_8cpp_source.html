128  //Base case:
129  if(root==NULL) return;
130  //If the data we want to remove is less than the current node's data...
131  else if(target<root->data){
132  //The node that will be deleted lies in our left subtree, so we
133  //recursively delete the node from our left subtree.
134  delNode(root->left,target);
135  }
136  //If the data we want to remove is greater than the current node's data...
137  else if(target>root->data){
138  //The node that will be deleted lies in our right subtree, so we
139  //recursively delete the node from our right subtree.
140  delNode(root->right,target);
141  }
142  //If we've found the node that will be deleted...
143  else{
144  //Case 1: If the node we need to delete has no children...
145  if(root->left==NULL&&root->right==NULL){
146  //We delete the node w/ our target value.
147  delete root;
148  //We set it to NULL to remove it from our tree.
149  root=NULL;
150  }
151  //Case 2: If the node we need to delete has 1 child.
152  else if(root->left==NULL){ //1 right child.
153  //We store the right child in a temp. variable.

246 void Tree::balance(Node *&root){
247  //If we have an invalid node, we simply exit.
248  if(root==NULL) return;
249  //Storing the heights of the left and right subtrees.
250  int lh=height(root->left);
251  int rh=height(root->right);
252  //If the tree is right heavy...
253  if(rh-lh>1){
254  //Storing the heights of the right subtree's left and right subtrees.
255  lh=height(root->right->left); //Right subtree's left subtree height.
256  rh=height(root->right->right); //Right subtree's right subtree height.
257  //If the right subtree is left heavy...
258  if(rh-lh<0) lftRghtRot(root); //Perform a left-right rotation.
259  //Otherwise...
260  else lftRot(root); //Perform a single left rotation.
261  }
262  //If the tree is left heavy...
263  else if(rh-lh<-1){
264  //Storing the heights of the left subtree's left and right subtrees.
265  lh=height(root->left->left); //Left subtree's left subtree height.
266  rh=height(root->left->right); //Left subtree's right subtree height.
267  //If the left subtree is right heavy...
268  if(rh-lh>0) rghtLftRot(root); //Perform a right-left rotation.
269  //Otherwise...

271  }
272  //NOTE: If the difference between the heights of the subtrees is

274  //see which rotation is needed to bring the tree back to a balanced state.
